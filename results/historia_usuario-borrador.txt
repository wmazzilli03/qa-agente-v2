Feature: Inicio de sesión de usuario

  @login_exitoso
  Scenario: Login exitoso redirige al inventario
    Given el usuario está en la página de login
    When ingresa usuario "standard_user" y contraseña "secret_sauce"
    And hace clic en el botón "Login"
    Then es redirigido al inventario de productos
    And visualiza el catálogo disponible

  @contraseña_incorrecta
  Scenario: Contraseña incorrecta muestra mensaje de error
    Given el usuario está en la página de login
    When ingresa usuario "standard_user" y contraseña "contraseña_incorrecta"
    And hace clic en el botón "Login"
    Then visualiza mensaje de error
    And no es redirigido al inventario

  @usuario_inexistente
  Scenario: Usuario inexistente muestra mensaje de error
    Given el usuario está en la página de login
    When ingresa usuario "usuario_no_existe" y contraseña "secret_sauce"
    And hace clic en el botón "Login"
    Then visualiza mensaje de error
    And no es redirigido al inventario

  @usuario_bloqueado
  Scenario: Usuario bloqueado no puede ingresar
    Given el usuario está en la página de login
    When ingresa usuario "locked_out_user" y contraseña "secret_sauce"
    And hace clic en el botón "Login"
    Then visualiza mensaje indicando usuario bloqueado
    And no es redirigido al inventario

  @campos_vacios
  Scenario: Campos vacíos muestra mensaje de validación
    Given el usuario está en la página de login
    When deja los campos de usuario y contraseña vacíos
    And hace clic en el botón "Login"
    Then visualiza mensaje de validación
    And no es redirigido al inventario

  @solo_usuario_ingresado
  Scenario: Solo usuario ingresado muestra mensaje de error
    Given el usuario está en la página de login
    When ingresa usuario "standard_user" y deja contraseña vacía
    And hace clic en el botón "Login"
    Then visualiza mensaje de error
    And no es redirigido al inventario

  @acceso_sin_autenticacion
  Scenario: Acceso directo sin autenticación redirige a login
    Given el usuario no está autenticado
    When intenta acceder directamente a la URL del inventario
    Then es redirigido a la página de login

  @flujo_repetido_login
  Scenario: Flujo repetido de login funciona correctamente
    Given el usuario está en la página de login
    When ingresa usuario "standard_user" y contraseña "secret_sauce"
    And hace clic en el botón "Login"
    And es redirigido al inventario de productos
    And cierra sesión
    And vuelve a la página de login
    And ingresa usuario "standard_user" y contraseña "secret_sauce"
    Then es redirigido al inventario de productos correctamente
